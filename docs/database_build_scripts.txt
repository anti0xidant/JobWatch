GO

USE JobWatch

CREATE TABLE Jobs
(
JobID 			int 			not null	 PRIMARY KEY,
CompanyID 		int 			not null,
Title 			nvar(30)		not null,
TypeID 			int not null,
Url 			nvarchar(2083),
LocationID 		int 			not null,
Description 		nvarchar(4000),
DateListed 		datetime,
DateApplied 		datetime 		not null,
IsActive 		bit 			not null,
IntermediatePhaseID 	int 			not null,
InterviewCount 		int 			not null,
OfferPhaseID 		int 			not null,
OfferedSalary 		smallmoney,
RejectionStatus 	bit 			not null
)

CREATE TABLE Company
(
CompanyID		int			not null	PRIMARY KEY,
CompanyName		nvar(30)		not null,
CompanyUrl		nvarchar(2083)	
)

CREATE TABLE Location
(
LocationID 		int 			not null	PRIMARY KEY,
City			nchar(30)		not null,
State			nchar(15)		not null
)

CREATE TABLE IntermediatePhase
(
IntermediatePhaseID	int			not null	PRIMARY KEY,
IntermediatePhase	nchar(25)		not null
)

CREATE TABLE OfferPhase
(
OfferPhaseID		int			not null 	PRIMARY KEY,
OfferPhase		nchar(25)		not null,
)

CREATE TABLE Language
(
LanguageID		int			not null	PRIMARY KEY,
Language		nchar(20)		not null
)

CREATE TABLE JobsxLanguage
(
JobID			int			not null	FOREIGN KEY,
LanguageID		int	 		not null	FOREIGN KEY
)
